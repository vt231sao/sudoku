/**
 * @module Store/Leaderboard
 * @description Zustand стейт-менеджер для управління таблицею лідерів. Відповідає за збереження, сортування та лімітування результатів гравців.
 */

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

/**
 * Модель результату однієї гри.
 * @typedef {Object} ScoreEntry
 * @property {string} [username] - Ім'я гравця.
 * @property {number} [score] - Кількість набраних балів.
 * @property {string} [difficulty] - Рівень складності.
 * @property {string} date - ISO-рядок дати та часу завершення гри.
 */

/**
 * @typedef {Object} LeaderboardState
 * @property {ScoreEntry[]} scores - Масив збережених результатів, відсортований за спаданням балів (максимум 20 записів).
 * @property {function(Object): void} addScore - Екшн для додавання нового результату. Автоматично сортує масив та залишає лише топ-20.
 */

/**
 * Хук для доступу до таблиці лідерів. Дані автоматично зберігаються в Local Storage.
 * @type {import('zustand').UseBoundStore<import('zustand').StoreApi<LeaderboardState>>}
 */
export const useLeaderboardStore = create(
    persist(
        (set) => ({
            scores: [],
            addScore: (newScore) => set((state) => {
                const updatedScores = [...state.scores, { ...newScore, date: new Date().toISOString() }];
                return {
                    scores: updatedScores.sort((a, b) => b.score - a.score).slice(0, 20)
                };
            }),
        }),
        {
            name: 'sudoku-leaderboard-storage',
        }
    )
);
