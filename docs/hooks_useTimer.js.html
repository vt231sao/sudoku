/**
 * @module Hooks/useTimer
 * @description Кастомний React хук для управління ігровим таймером.
 */

import { useState, useEffect, useRef } from 'react';

/**
 * Об'єкт, що містить стан таймера та функції для управління ним.
 * @typedef {Object} TimerState
 * @property {number} time - Поточний час у секундах.
 * @property {boolean} isActive - Чи запущений таймер в даний момент.
 * @property {function(): void} startTimer - Функція для запуску таймера.
 * @property {function(): void} stopTimer - Функція для зупинки таймера.
 * @property {function(): void} resetTimer - Функція для скидання часу до нуля.
 */

/**
 * Хук `useTimer`.
 * @param {boolean} [autoStart=false] - Визначає, чи повинен таймер запускатися автоматично при монтуванні компонента.
 * @returns {TimerState} Об'єкт стану та методів таймера.
 */
export function useTimer(autoStart = false) {
    const [time, setTime] = useState(0);
    const [isActive, setIsActive] = useState(autoStart);
    const intervalRef = useRef(null);

    function startTimer() {
        setIsActive(true);
    }

    function stopTimer() {
        setIsActive(false);
    }

    function resetTimer() {
        setTime(0);
        setIsActive(autoStart);
    }

    useEffect(() => {
        if (isActive) {
            intervalRef.current = setInterval(() => {
                setTime(t => t + 1);
            }, 1000);
        } else if (!isActive && intervalRef.current) {
            clearInterval(intervalRef.current);
            intervalRef.current = null;
        }

        return () => {
            if (intervalRef.current) {
                clearInterval(intervalRef.current);
            }
        };
    }, [isActive]);

    return { time, startTimer, stopTimer, resetTimer, isActive };
}
